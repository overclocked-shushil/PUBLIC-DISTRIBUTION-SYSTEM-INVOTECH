<script>

let currentScreen = 'home';

let timerInterval = null;


function navigate(screen) {

  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));

  document.getElementById('screen-' + screen).classList.add('active');

  currentScreen = screen;

  document.getElementById('navBack').classList.toggle('visible', screen !== 'home');

  window.scrollTo(0, 0);


  if (screen === 'cust-record') {

    document.getElementById('savedDate').textContent = new Date().toLocaleDateString('en-IN', {day:'numeric',month:'short',year:'numeric'});

  }

  if (screen === 'complaint-done') {

    document.getElementById('complaintDate').textContent = new Date().toLocaleDateString('en-IN', {day:'numeric',month:'short',year:'numeric'});

  }

}


function goHome() {

  navigate('home');

}


function sendOTP(type) {

  let phone = '';

  if (type === 'admin') phone = document.getElementById('adminPhone').value || '+91 98765 43210';

  if (type === 'dist')  phone = document.getElementById('distPhone').value || '+91 98765 43210';

  if (type === 'cust')  phone = document.getElementById('custPhone').value || '+91 98765 43210';

  navigate(type + '-otp');

  if (type === 'admin') startTimer('adminTimer');

}


function verifyOTP(type) {

  // Collect OTP

  let prefix = type === 'admin' ? 'ao' : (type === 'dist' ? 'do' : 'co');

  let otp = '';

  for (let i = 1; i <= 6; i++) {

    let el = document.getElementById(prefix + i);

    if (el) otp += el.value;

  }

  if (otp.length < 6) {

    alert('Please enter all 6 digits. Use demo OTP: 123456');

    return;

  }

  if (otp !== '123456') {

    alert('Incorrect OTP! Use demo OTP: 123456');

    return;

  }

  if (type === 'admin') navigate('admin-dashboard');

  if (type === 'dist')  navigate('ration');

  if (type === 'cust')  navigate('family');

}


function submitComplaint() {

  const name = document.getElementById('pubName').value;

  if (!name.trim()) { alert('Please enter your name'); return; }

  navigate('complaint-done');

}


function otpMove(current, nextId) {

  if (current.value.length === 1 && nextId) {

    const next = document.getElementById(nextId);

    if (next) next.focus();

  }

}


function startTimer(id) {

  let seconds = 120;

  if (timerInterval) clearInterval(timerInterval);

  timerInterval = setInterval(() => {

    const el = document.getElementById(id);

    if (!el) { clearInterval(timerInterval); return; }

    seconds--;

    const m = Math.floor(seconds / 60);

    const s = seconds % 60;

    el.textContent = m + ':' + String(s).padStart(2, '0');

    if (seconds <= 0) {

      clearInterval(timerInterval);

      el.textContent = 'Expired';

    }

  }, 1000);

}
